```python
class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:
        l,r =1, len(nums)-1
        numss = sorted(nums)
        n = 0
        output = []
        
        for i in range(r -1):
            l,r = i+1, len(nums)-1
            while l < r:
                if numss[i] + numss[l] + numss[r] ==0:
                    output.insert(n, [numss[i],numss[l],numss[r]])
                    n += 1
                    l += 1
                elif numss[i] + numss[l] + numss[r] >0:
                    l += 1
                else:
                    r -= 1
            continue
        return output
```

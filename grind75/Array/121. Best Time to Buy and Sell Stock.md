class Solution {
    public int maxProfit(int[] prices) {
        int i = 0;
        int j = 1;
        int maxProfit = 0;
        for(i= 0; i <= prices.length-2; i++){
            for(j=i+1; j <= prices.length-1 && prices[j] >prices[i]; j++){
                int tempProfit = prices[j] - prices[i];
                if(tempProfit > maxProfit){
                    maxProfit = tempProfit;
                }
            }
        }
        return maxProfit;
    }
}
